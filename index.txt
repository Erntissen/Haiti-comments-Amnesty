    <script>
        function copyText(id) {
            const text = document.getElementById(id).innerText;
            navigator.clipboard.writeText(text).then(function() {
                alert("Text copied to clipboard!");
            }, function(err) {
                alert("Error copying text: " + err);
            });
        }
